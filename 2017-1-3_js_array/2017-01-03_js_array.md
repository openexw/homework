## 2.数组[4,5,6,4,5,7,8,9,7,6,5], 去掉数组中重复的元素。

```
var b = [4,5,6,4,5,7,8,9,7,6,5];
    var tmp = [];
    function s(b) {
        for (var i in b) {
            if (tmp.indexOf(b[i] == -1)) {
                tmp.push(b[i]);
            }
        }
        return tmp;
    }
```
